Send a Security Alert

curl -X POST https://autodocmvp.vercel.app/v1/fix/create \
  -H "Content-Type: application/json" \
  -d '{
    "alert_id": "CVE-2024-12345",
    "repository": "your-org/your-repo",
    "alert_source": "Dependabot",
    "alert_data": {
      "vulnerability": "Prototype Pollution in lodash",
      "severity": "high",
      "affected_version": "<4.17.20",
      "fixed_version": "4.17.21",
      "dependency": "lodash"
    }
  }'
